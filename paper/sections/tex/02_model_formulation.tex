\section{モデルの定式化}

\subsection{半マルコフ過程}

半マルコフ過程は，遷移先は現在状態のみに依存するが，各状態での
滞在時間には指数分布に限らない任意の分布を許す確率過程である
\cite{limnios2001,cinlar1975,howard1971}．本稿では，潜伏時間に
Pareto 分布を直接導入するため，この枠組みを用いる．
全損または復旧完了は一つのインシデント経路の終端であり，その時点で
当該インシデントの費用を計上する．

\subsection{攻撃フェーズと防疫フェーズ}

\begin{definition}[攻撃フェーズ]
状態空間を $\{A_0,A_1,A_2,A_3\}$ とする．
$A_0$ は無感染，$A_1$ は潜伏感染，$A_2$ は発症中，
$A_3$ は破壊完了を表す．攻撃到着は強度 $\lambda>0$ の
Poisson 過程であり，攻撃到着時に $A_0\to A_1$ が起こる．
潜伏時間 $T_d$ の経過後に $A_1\to A_2$ が起こる．発症後の結果は，
防疫フェーズで定義する症状検知分岐により，隔離・復旧または
$A_3$ への全損として扱う．
\end{definition}

\begin{assumption}[Pareto 潜伏時間]
$T_d\sim\Pareto(t_m,\alpha(I))$ とし，
\[
  \Prob(T_d>t)=\left(\frac{t_m}{t}\right)^{\alpha(I)},\qquad
  f_{T_d}(t)=\alpha(I)t_m^{\alpha(I)}t^{-(\alpha(I)+1)}
\]
for $t\ge t_m>0$ とする．また $0\le t<t_m$ では
$\Prob(T_d>t)=1$ である．尾指数 $\alpha(I)>0$ は投資 $I$ の関数である．
\end{assumption}

\begin{definition}[防疫フェーズ]
状態空間を $\{D_0,D_1,D_2,D_3\}$ とする．
$D_0$ は通常，$D_1$ は検知，$D_2$ は隔離，$D_3$ は復旧中を表す．
早期検知時刻は $T_{\mathrm{det}}\sim\Exp(\beta(I))$ であり，
$T_d$ とは独立とする．$T_{\mathrm{det}}<T_d$ の場合に早期検知が
生起する．発症後には確率 $\eta\in(0,1]$ で症状検知され，
症状検知に失敗した場合は全損として費用 $L$ を計上する．
\end{definition}

\begin{assumption}[投資応答関数]
$\alpha(I)$ と $\beta(I)$ は $[0,\infty)$ 上の $C^2$ 関数で，
単調増加かつ凹型である．また
\[
  \lim_{I\to\infty}\alpha(I)=\alpha_{\max}<\infty,\qquad
  \lim_{I\to\infty}\beta(I)=\beta_{\max}<\infty,
\]
かつ $\alpha(0)>0,\ \beta(0)>0$ とする．
\end{assumption}

合成状態空間は
\[
S=\{S_N,S_L,S_E,S_S,S_I,S_R,S_F\}
\]
で表す．$S_N$ は正常，$S_L$ は潜伏・未検知，$S_E$ は早期検知済，
$S_S$ は発症後・症状検知待ち，$S_I$ は隔離済，$S_R$ は復旧中，
$S_F$ は全損を表す．

\subsection{バックアップと復旧}

\begin{definition}[バックアップスケジュール]
間隔 $\Delta>0$ で $n\in\mathbb{Z}_{\ge1}$ 世代のバックアップを
保持する．侵入後に取得されたバックアップは汚染されるとし，
実効保持期間を
\[
  a_n(\Delta)=n\Delta
\]
とする．感染後経過時間 $T_{\mathrm{age}}$ が
$T_{\mathrm{age}}>a_n(\Delta)$ なら清浄なバックアップが保持期間外に
押し出され，全損として扱う．$T_{\mathrm{age}}\le a_n(\Delta)$ なら
復旧可能である．汚染世代数を整数で記録する場合は
$k=\lfloor T_{\mathrm{age}}/\Delta\rfloor$ と見れば，この判定は
零確率の境界を除いて $k<n$ と同値である．Pareto 尾確率を
保持期間で評価する議論では
\[
  \Delta\in\mathcal{D}_n:=\{\Delta>t_m/n\}
\]
を設計領域とする．
\end{definition}

\begin{definition}[コンテナベース復旧時間]
復旧は隔離，クローン，再デプロイの三段階から成る．
$T_{\mathrm{iso}}\sim\Exp(\mu_{\mathrm{iso}})$，
$T_{\mathrm{clone}}\sim\Exp(\mu_c)$，
$T_{\mathrm{deploy}}\sim\Exp(\mu_d)$ とし，
\[
  \tau_R:=\mu_{\mathrm{iso}}^{-1}+\mu_c^{-1}+\mu_d^{-1}.
\]
感染後経過時間を $T_{\mathrm{age}}$ と書く．早期検知分岐では
$T_{\mathrm{age}}=T_{\mathrm{det}}$，症状検知分岐では
$T_{\mathrm{age}}=T_d$ である．スキャンに伴う追加停止時間は
$T_{\mathrm{scan}}=\kappa_sT_{\mathrm{age}}$ とする．ここで
$\kappa_s\ge0$ は無次元のスキャン時間係数である．第 5 節で閉形式の
内点最適間隔を扱うときは，保持期間を延ばすことに正の限界停止時間
費用が生じる場合として $\kappa_s>0$ を仮定する．
\end{definition}

\subsection{期待損失と費用レート}

費用記号を次のように定める．$d_1$ は停止一日あたり損失，
$d_0$ は復旧に伴う固定費，$L$ は全損時損失，$c_b$ は一回の
バックアップ固定費，$c_I$ は投資一単位あたり費用，$c_n$ は
追加保持一世代あたり費用である．

投資 $I$ と保持世代数 $n$ を固定し，
$\alpha=\alpha(I),\ \beta=\beta(I)$ と略記する．早期検知に失敗する
確率は
\[
  \Lbeta:=\E[e^{-\beta T_d}]=1-p_e,\qquad
  p_e=\Prob(T_{\mathrm{det}}<T_d)
\]
である．保持期間 $a(\Delta)=n\Delta$ に対し，
\begin{align}
  \Hbeta(\Delta;\alpha)
  &:=\E[e^{-\beta T_d}\mathbf{1}_{\{T_d>a(\Delta)\}}],\\
  \Mbeta(\Delta;\alpha)
  &:=\E[T_d e^{-\beta T_d}\mathbf{1}_{\{T_d\le a(\Delta)\}}].
\end{align}
ここで $\Hbeta$ は「早期検知に失敗し，保持期間も超過する」確率，
$\Lbeta-\Hbeta$ は「早期検知には失敗したが保持期間内で症状検知
可能な」分岐確率である．

\begin{assumption}[早期検知費用]
早期検知に成功した分岐の期待費用を $\Qearly(I)$ と書く．
各固定投資 $I$ について $\Qearly(I)<\infty$ であり，バックアップ間隔
$\Delta$ には依存しないとする．
\end{assumption}

本稿では，早期検知時の封じ込め・隔離費用は $\Delta$ に依存しない
ものとして分離する．
早期検知に失敗した攻撃から生じる，バックアップ間隔依存の残余期待
損失を
\begin{equation}
  \Qfull(\Delta;\alpha,\beta)
  =\eta(d_1\tau_R+d_0)(\Lbeta-\Hbeta)
  +\eta d_1\kappa_s\Mbeta
  +L\{(1-\eta)(\Lbeta-\Hbeta)+\Hbeta\}
  \label{eq:qfull}
\end{equation}
と定義する．

費用レート \eqref{eq:cost-rate} は，インシデントが強度 $\lambda$ の
Poisson 過程で発生し，各インシデントに独立な半マルコフ経路マークが
付くという暦時間のリスク会計として読む．すなわち，状態図は一つの
インシデント内の経路を表し，復旧中にシステム到着過程を停止させる
再生サイクル比ではない．この会計では，Poisson 報酬過程の平均レート
から定常費用レートは
\begin{equation}
  C(\Delta,I,n_{\mathrm{ext}})
  =\lambda\{\Qearly(I)+\Qfull(\Delta;\alpha(I),\beta(I))\}
  +\frac{c_b}{\Delta}+c_I I+c_n n_{\mathrm{ext}}.
  \label{eq:cost-rate}
\end{equation}
$\Delta$ 最適化では $\Qearly(I)$，$c_I I$，$c_n n_{\mathrm{ext}}$ は
定数として扱う．
